								-- –ü—Ä–æ–≤–µ—Ä–∫–∞ —Ç–∏–ø–æ–≤ —Å–æ–±—ã—Ç–∏–π –¥–ª—è –∫–∞–º–µ—Ä
								event_types_base_check AS (
									SELECT 
										COUNT(*) > 0 AS has_event_types
									FROM event_types et
									WHERE et.origin_class = 'Device::Camera' 
										AND et.filter_name IS NOT NULL
										AND et.deleted_at IS NULL
								),
								-- –ü—Ä–æ–≤–µ—Ä–∫–∞ –≤–æ–∑–º–æ–∂–Ω–æ—Å—Ç–µ–π –∫–∞–º–µ—Ä –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
								camera_capabilities AS (
									SELECT 
										BOOL_OR(COALESCE((data->'caps'->>'motion_detection')::boolean, false)) AS has_motion_detection,
										BOOL_OR(COALESCE((data->'caps'->>'sound_detection')::boolean, false)) AS has_sound_detection,
										BOOL_OR(COALESCE((data->'caps'->>'line_crossing_detection')::boolean, false)) AS has_line_crossing,
										BOOL_OR(COALESCE((data->'caps'->>'face_detection')::boolean, false)) AS has_face_detection,
										BOOL_OR(COALESCE((data->'caps'->>'tampering_detection')::boolean, false)) AS has_tampering,
										BOOL_OR(COALESCE((data->'caps'->>'human_detection')::boolean, false)) AS has_human_detection
									FROM cameras 
									WHERE user_id = :user_id AND deleted_at IS NULL
								),
								-- –°–æ–±–∏—Ä–∞–µ–º —Ç–∏–ø—ã —Å–æ–±—ã—Ç–∏–π –¥–ª—è –∫–∞–º–µ—Ä
								event_types_data AS (
									SELECT 
										et.id,
										et.name
									FROM event_types et
									WHERE et.origin_class = 'Device::Camera' 
										AND et.filter_name IS NOT NULL
										AND et.deleted_at IS NULL
										AND (
											(et.filter_name = 'motion_detection' AND (SELECT has_motion_detection FROM camera_capabilities)) OR
											(et.filter_name = 'sound_detection' AND (SELECT has_sound_detection FROM camera_capabilities)) OR
											(et.filter_name = 'line_crossing_detection' AND (SELECT has_line_crossing FROM camera_capabilities)) OR
											(et.filter_name = 'face_detection' AND (SELECT has_face_detection FROM camera_capabilities)) OR
											(et.filter_name = 'tampering_detection' AND (SELECT has_tampering FROM camera_capabilities)) OR
											(et.filter_name = 'human_detection' AND (SELECT has_human_detection FROM camera_capabilities)) OR
											et.filter_name IN ('camera_offline', 'camera_online')
										)
								)

								-- –§–∏–Ω–∞–ª—å–Ω—ã–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç
								SELECT 
									-- –ì—Ä—É–ø–ø—ã —Å–æ–±—ã—Ç–∏–π
									json_build_object(
										'enabled', (SELECT groups_enabled FROM groups_check),
										'content', CASE 
											WHEN (SELECT groups_enabled FROM groups_check) THEN 
												(SELECT COALESCE(json_agg(json_build_object(
													'id', gd.id,
													'name', gd.name,
													'description', gd.description
												)), '[]'::json) FROM groups_data gd)
											ELSE NULL
										END
									) AS groups,
									
									-- –£—Å—Ç—Ä–æ–π—Å—Ç–≤–∞
									json_build_object(
										'enabled', (SELECT devices_enabled FROM devices_check)
									) AS devices,
									
									-- –¢–∏–ø—ã —Å–æ–±—ã—Ç–∏–π
									json_build_object(
										'enabled', (
											(SELECT has_event_types FROM event_types_base_check) AND 
											(SELECT has_cameras FROM cameras_check)
										),
										'content', CASE 
											WHEN (SELECT has_event_types FROM event_types_base_check) AND 
													(SELECT has_cameras FROM cameras_check) THEN
												(SELECT COALESCE(json_agg(json_build_object(
													'id', etd.id,
													'name', etd.name
												)), '[]'::json) FROM event_types_data etd)
											ELSE NULL
										END
									) AS event_types;
